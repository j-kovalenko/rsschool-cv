def caesar_cipher(word, shift=3):
result = ''
for i in range(0, len(word)):
 if ord('A') <= ord(word[i]) <= ord('Z'):
  char = ((ord(word[i]) + shift - ord('A')) % 32) + ord('A')
  result += chr(char)
 elif ord('a') <= ord(word[i]) <= ord('z'):
  char = ((ord(word[i]) + shift - ord('a')) % 32) + ord('a')
  result += chr(char)
 else:
  result += word[i]
 return result
